programa

{


	funcao inicio()

	{

      inteiro opcao = 1

      real total_pedido = 0.0



      enquanto (opcao != 4)

      {

      escreva("\n==== TOTEM DE ATENDIMENTO ====\n")

      escreva("Opção 1: Hambúrguer (R$25,00)\n")

      escreva("Opção 2: Batata frita (R$15,00)\n")

      escreva("Opção 3: Refrigerante (R$10,00)\n")

      escreva("Opção 4: Finalizar pedido \n")

      escreva("Escolha uma opção: ")

      leia(opcao)



      escolha (opcao)

      {

      	caso 1:

      	total_pedido = total_pedido + 25

      	escreva ("Hambúrguer adicionado ao carrinho!\n")

      	pare

      	caso 2:

      	total_pedido = total_pedido + 15

      	escreva ("Batata frita adicionada ao carrinho!\n")

      	pare

      	caso 3: 

      	total_pedido = total_pedido + 10

      	escreva ("Refrigerante adicionado ao pedido!\n")

      	pare

      	caso 4:

      	escreva ("\nFinalizando pedido...\n")

      	pare



      	caso contrario:

      	escreva ("Opção inválida! Tente novamente.\n")

      }

      escreva("Total parcial: R$ ",total_pedido,"\n")

      }

      //exibição do valor total

      escreva("\n===== NOTA FISCAL =====\n")

      escreva("Valor total da compra: R$ ", total_pedido,"\n")



      //simulação de impressão do cupom

       para(inteiro contador = 1; contador <= 5; contador++)

        {

            escreva("Imprimindo cupom fiscal... ", contador, "\n")

        }




	escreva("Sistema encerrado.\n")

      	

      }


}
